首先需要处理下pcm，从32位深转为16位深，因为max98357只支持16位：

```python
# convert_pcm_32_to_16.py
import struct
import sys

def convert_pcm_32_to_16(input_file, output_file):
    with open(input_file, 'rb') as fin, open(output_file, 'wb') as fout:
        while True:
            # 每次读取 4096 字节（1024 个 32-bit 样本）
            raw = fin.read(4096)
            if not raw:
                break
            
            # 解析为 int32 数组（小端序）
            num_samples = len(raw) // 4
            samples_32 = struct.unpack(f'<{num_samples}i', raw)
            
            # 转为 16-bit（饱和处理防止溢出）
            samples_16 = []
            for s in samples_32:
                if s > 0x7FFF0000:
                    s16 = 0x7FFF
                elif s < -0x80000000:
                    s16 = -0x8000
                else:
                    s16 = s >> 16  # 等效于 s // 65536
                samples_16.append(s16)
            
            # 写入 16-bit 小端序 PCM
            fout.write(struct.pack(f'<{len(samples_16)}h', *samples_16))

    print(f"Converted {input_file} -> {output_file}")

if __name__ == "__main__":
    if len(sys.argv) != 3:
        print("Usage: python convert_pcm_32_to_16.py input.pcm output.pcm")
        sys.exit(1)
    
    convert_pcm_32_to_16(sys.argv[1], sys.argv[2])
```

服务端代码：
```python
# pcm_server.py （增强版：带音量放大）
import socket
import struct
import os

PCM_FILE = "r16.pcm"   # 16kHz, 16-bit, mono, little-endian
HOST = "0.0.0.0"
PORT = 8888
CHUNK_SAMPLES = 1024

# ===== 音量增益（关键！）=====
GAIN = 8.0  # 尝试 2.0, 4.0, 8.0... 但不要超过 32767（避免削波失真）

def main():
    if not os.path.exists(PCM_FILE):
        print(f"Error: {PCM_FILE} not found!")
        return

    with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as server_sock:
        server_sock.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
        server_sock.bind((HOST, PORT))
        server_sock.listen(1)
        print(f"Server listening on {HOST}:{PORT} (Gain={GAIN})")

        while True:
            conn, addr = server_sock.accept()
            print(f"Client connected from {addr}")
            try:
                with open(PCM_FILE, "rb") as f:
                    while True:
                        raw = f.read(CHUNK_SAMPLES * 2)
                        if not raw:
                            break

                        num = len(raw) // 2
                        samples_mono = struct.unpack(f"<{num}h", raw)

                        samples_stereo = []
                        for s in samples_mono:
                            # 应用增益
                            s_amp = int(s * GAIN)
                            # 饱和处理：防止溢出（削波）
                            if s_amp > 32767:
                                s_amp = 32767
                            elif s_amp < -32768:
                                s_amp = -32768
                            samples_stereo.extend([s_amp, s_amp])

                        data_out = struct.pack(f"<{len(samples_stereo)}h", *samples_stereo)
                        conn.sendall(data_out)
                print("Finished sending")
            except Exception as e:
                print(f"Error: {e}")
            finally:
                conn.close()

if __name__ == "__main__":
    main()
```

其中GAIN可以调，放大音量。

```c
#include <WiFi.h>
#include "driver/i2s.h"

// ===== 配置 =====
// Wi-Fi 配置
const char* ssid = "xxxx";
const char* password = "xxxxxxxx";

// Python 服务端地址
const char* serverIP = "192.168.31.159";  // 替换为你的电脑 IP
const uint16_t serverPort = 8888;

// I2S 引脚（MAX98357A）
#define I2S_WS 25   // LRCLK
#define I2S_SD 32   // DIN
#define I2S_SCK 33  // BCLK

#define SAMPLE_RATE 16000
#define I2S_PORT I2S_NUM_0
#define RX_BUFFER_SIZE 2048  // 必须是 4 的倍数（stereo 16-bit = 4 bytes/sample）

uint8_t rxBuffer[RX_BUFFER_SIZE];
WiFiClient client;

void setup() {
  Serial.begin(115200);

  // Connect to Wi-Fi
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\nWiFi connected");

  // I2S Config
  i2s_config_t i2s_config = {
    .mode = (i2s_mode_t)(I2S_MODE_MASTER | I2S_MODE_TX),
    .sample_rate = SAMPLE_RATE,
    .bits_per_sample = I2S_BITS_PER_SAMPLE_16BIT,
    .channel_format = I2S_CHANNEL_FMT_RIGHT_LEFT,
    .communication_format = (i2s_comm_format_t)I2S_COMM_FORMAT_STAND_I2S,
    .dma_buf_count = 8,
    .dma_buf_len = 64,
    .use_apll = false,
    .tx_desc_auto_clear = true
  };

  i2s_pin_config_t pin_config = {
    .bck_io_num = I2S_SCK,
    .ws_io_num = I2S_WS,
    .data_out_num = I2S_SD,
    .data_in_num = I2S_PIN_NO_CHANGE
  };

  i2s_driver_install(I2S_PORT, &i2s_config, 0, NULL);
  i2s_set_pin(I2S_PORT, &pin_config);
  i2s_zero_dma_buffer(I2S_PORT);
}

void playFromServer() {
  if (!client.connect(serverIP, serverPort)) {
    Serial.println("Failed to connect to server");
    return;
  }
  Serial.println("Connected to server, playing...");

  while (client.connected()) {
    size_t avail = client.available();
    if (avail == 0) {
      delay(1);
      continue;
    }

    size_t toRead = (avail > RX_BUFFER_SIZE) ? RX_BUFFER_SIZE : avail;
    toRead = (toRead / 4) * 4; // 对齐到 4 字节（stereo 16-bit）
    if (toRead == 0) continue;

    int bytesRead = client.read(rxBuffer, toRead);
    if (bytesRead <= 0) break;

    size_t bytesWritten = 0;
    i2s_write(I2S_PORT, rxBuffer, bytesRead, &bytesWritten, portMAX_DELAY);
  }

  client.stop();
  i2s_zero_dma_buffer(I2S_PORT);
  Serial.println("Playback done");
}

void loop() {
  playFromServer();
  delay(300000); // 3秒后重试
}
```
